  <script>
    function saveNote() {
      localStorage.setItem("note", document.getElementById("note").value);
      alert("Note saved!");
    }

    function clearNote() {
      document.getElementById("note").value = "";
    }

    window.onload = function () {
      document.getElementById("note").value =
        localStorage.getItem("note") || "";
    };
